-- B. JSONB ==============================================

-- 1. создать таблицу и внести данные
create table jsonb_test (id serial, filename varchar(255), data jsonb);

insert into jsonb_test (filename, data) values
	('testfile.csv', '{"ias5_d": "10670100", "acnt_gid": "28443939595539", "adj_code": "LOANS_TEST"}'),
	('testfile.csv', '{"ias5_d": "1067023242", "acnt_gid": "2844393953242", "adj_code": "VTB_TEST" }'),
	('testfile.csv', '{"ias5_d": "10245200", "acnt_gid": "284439323423439"}');


-- 2. различные выборки
select * from jsonb_test;
select data->>'acnt_gid' as acnt_gid from jsonb_test;
select * from jsonb_test where data->>'adj_code' = 'LOANS_TEST';
select jsonb_pretty(data) from jsonb_test where data->>'adj_code' = 'LOANS_TEST';

-- 3. использование json как колонок
select t.filename,
		t.id,
		js.ias5_d,
		js.acnt_gid,
		js.adj_code
from jsonb_test t
cross join lateral jsonb_to_record(data) as js(ias5_d varchar(100), acnt_gid varchar(100), adj_code varchar(20));

-- 4. пример выдачи объектов с jsonb 
select json_build_object('user_id', t.id, 'name', t.filename) from jsonb_test t;
select jsonb_agg(json_build_object('user_id', t.id, 'name', t.filename)) from jsonb_test t;

SELECT jsonb_pretty(jsonb_build_object(
		'user_id', u.id, 
		'name', u.name, 
		'orders', jsonb_agg(
			jsonb_build_object(
	       		'order_id', o.id, 
	       		'amount', o.amount)
       		)
	))
	FROM json_users u
	JOIN json_orders o on u.id = o.user_id 
	WHERE u.id = 2
	group by u.id;

-- C. ПОЛНОТЕКСТОВЫЙ ПОИСК ==============================================

-- 1. после импорта таблицы посмотреть содержимое
SELECT id, parent_id, sent, subject, author, left(body_plain,400) FROM mail_messages LIMIT 1;
SELECT count(*) FROM mail_messages;

-- 2. полнотекстовый поиск
SELECT id, subject FROM mail_messages WHERE to_tsvector(subject) @@ to_tsquery('magical') ORDER BY id LIMIT 5;

SELECT id, 
		ts_rank(to_tsvector(subject), to_tsquery('magic & value')) as rank,
		subject, 
		to_tsvector(subject) as vector,  
		to_tsquery('magic & value') as query
FROM mail_messages WHERE to_tsvector(subject) @@ to_tsquery('magic & value') ORDER BY id LIMIT 5;

SELECT id, subject FROM mail_messages
WHERE to_tsvector(subject) @@ to_tsquery('magic & value & !time')
ORDER BY id LIMIT 5;

SELECT id, subject FROM mail_messages
WHERE to_tsvector(subject) @@ to_tsquery('magic & (value | constant)')
ORDER BY id LIMIT 5;

SELECT id, translate(
  ts_headline(body_plain,
              to_tsquery('magic'),
              'StartSel=*,StopSel=*,MinWords=8,MaxWords=10'),
  E'\n',
  ' '
) FROM mail_messages
WHERE to_tsvector(body_plain) @@ to_tsquery('magic')
ORDER BY id LIMIT 5;

-- 3. пример парсинга
SELECT * FROM ts_parse('default', 'The bells from the chapel went jingle-jangle');

-- 4. добавление индекса для быстрого поиска
ALTER TABLE mail_messages ADD search_vector tsvector GENERATED ALWAYS AS (to_tsvector('english',subject) || to_tsvector('english',body_plain)) STORED;
CREATE INDEX ON mail_messages USING gin(search_vector);
EXPLAIN (costs off) SELECT count(*) FROM mail_messages WHERE search_vector @@ to_tsquery('magic');

select id, body_plain from mail_messages where search_vector @@ to_tsquery('epstein');
