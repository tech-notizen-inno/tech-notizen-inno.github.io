-- 6. online операции с партициями 
-- исходная таблица
create table original_table (id int, t text);
insert into original_table (id, t) select floor(random() * (11))::int, random()::text from generate_series(1,30000000);

select min(id), max(id) from original_table;

-- секционированная таблица
create table part_table (id int, t text) partition by list(id);

create table part_table_123 partition of part_table for values in (1,2,3);
create table part_table_456 partition of part_table for values in (4,5,6);
create table part_table_789 partition of part_table for values in (7,8,9);
create table part_table_10000 partition of part_table for values in (10000);
create table part_table_default partition of part_table default;

-- триггер для обновления обоих таблиц в момент "перелива" данных
create or replace function part_update() returns trigger
as $$
begin
	if TG_OP = 'INSERT' or TG_OP = 'UPDATE' then
		insert into part_table (id, t) values (new.id, new.t);
		return new;
	elsif TG_OP = 'UPDATE' then
		update part_table set t = new.t where id=old.id;
		return new;
	elsif TG_OP 'DELETE' then
		delete from part_table where id = old.id;
		return old;
	end if;
end;
$$ language plpgsql;


create trigger trigger_part_table_upd
after insert or update or delete on original_table
for each row execute function part_update();

-- процедура для "перелива данных"

create or replace procedure pump_date()
as $$
declare 
	rec record;
begin
	for rec in
		select * from original_table
	loop
		insert into part_table (id, t) values (rec.id, rec.t);
	end loop;

	commit;
end;
$$ language plpgsql;

call pump_date();

-- 2. заполняем данными, проверяем, вставляем регион
insert into pgbench_accounts_to_div select pa.*, floor(random() * 100)::int as new_rnd_key from pgbench_accounts pa;
select min(new_rnd_key), max(new_rnd_key) from pgbench_accounts_to_div;
select new_rnd_key, count(*) from pgbench_accounts_to_div group by new_rnd_key order by new_rnd_key;
