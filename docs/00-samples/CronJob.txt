---
apiVersion: batch/v1beta1
kind: CronJob
metadata:
  name: hello-with-cron
spec:
  schedule: "*/1 * * * *"
  concurrencyPolicy: Allow
  jobTemplate:
    spec:
      backoffLimit: 2
      activeDeadlineSeconds: 100
      template:
        spec:  
          containers:
          - name: testcronjob
            image: busybox
            args:
              - /bin/sh
              - -c
              -  date; echo Hello CronJob!
          restartPolicy: Never
